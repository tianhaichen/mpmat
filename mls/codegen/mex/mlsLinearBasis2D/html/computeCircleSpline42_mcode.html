1function [w] = computeCircleSpline(x,xI,d,form)
 2% Compute cubic and quartic spline function
 3% Inputs:
 4% x (1x2)  : coordinate of point at which w is to be evaluated
 5% xI (1x2) : coord of node I
 6% d        : size of the support
 7
 8r = norm( x - xI )/d;
 9
10switch form
11  case 'cubic_spline' 
12     [w,~] = cubicSpline(r);
13  case 'quartic_spline'
14     [w,~] = quarticSpline(r);
15  otherwise 
16     error('Grr. Unknown functional form');
17end
