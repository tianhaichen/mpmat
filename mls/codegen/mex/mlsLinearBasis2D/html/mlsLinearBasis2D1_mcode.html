1function [phi] = mlsLinearBasis2D(pt,index,node,di,form)
 2%#codegen
 3% Compute the MLS shape function at point pt for all nodes within the
 4% support of this point pt.
 5% Basis used is linear basis pT = [1 x y]
 6%
 7% Vinh Phu Nguyen
 8% The University of Adelaide, Australia
 9% 9 October 2015.
10
11% --------------------------------------
12%      compute the moment matrix A
13% --------------------------------------
14nn   = length(index);
15A    = zeros(3,3);
16w    = zeros(nn,1);
17phi  = zeros(nn,1);
18
19for m = 1:nn
20  idm  = index(m);
21  xi   = node(idm,1);
22  yi   = node(idm,2);
23  wi   = computeCircleSpline(pt,[xi yi],di(idm),form);
24  pTp  = [1 xi yi;xi xi^2 xi*yi;yi xi*yi yi^2] ;
25  A    = A    + wi*pTp ;
26  w(m) = wi;
27end
28
29p  = [1; pt(1); pt(2)];
30
31c=A\p;
32           
33
34for m = 1:nn
35  xi  = node(index(m),:);
36  piT = [1; xi(1); xi(2)];
37  phi(m) = c'*piT*w(m) ;
38end
